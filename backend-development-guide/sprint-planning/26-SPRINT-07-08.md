```java
public class WebSocketConfig implements WebSocketMessageBrokerConfigurer {

    @Override
    public void configureMessageBroker(MessageBrokerRegistry config) {
        // Enable simple broker for /topic and /queue
        config.enableSimpleBroker("/topic", "/queue");

        // Application destination prefix
        config.setApplicationDestinationPrefixes("/app");

        // User destination prefix for private messages
        config.setUserDestinationPrefix("/user");
    }

    @Override
    public void registerStompEndpoints(StompEndpointRegistry registry) {

            .setAllowedOrigins("https://meslektas.com")
            .withSockJS();
    }
```

```java
public class WebSocketAuthenticationInterceptor implements ChannelInterceptor {

    @Override
    public Message<?> preSend(Message<?> message, MessageChannel channel) {
        StompHeaderAccessor accessor =
            MessageHeaderAccessor.getAccessor(message, StompHeaderAccessor.class);

        if (StompCommand.CONNECT.equals(accessor.getCommand())) {
            String token = accessor.getFirstNativeHeader("Authorization");
            // Validate JWT token
            // Set user in accessor
            accessor.setUser(userPrincipal);
        }

        return message;
```

```sql
CREATE TABLE conversations (
    id UUID PRIMARY KEY,
    participant1_id UUID NOT NULL REFERENCES users(id),
    participant2_id UUID NOT NULL REFERENCES users(id),
    last_message_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT unique_conversation UNIQUE(participant1_id, participant2_id),
    CONSTRAINT participants_different CHECK(participant1_id != participant2_id)
);

CREATE INDEX idx_conversations_participant1 ON conversations(participant1_id);
CREATE INDEX idx_conversations_participant2 ON conversations(participant2_id);
CREATE INDEX idx_conversations_last_message ON conversations(last_message_at DESC);

CREATE TABLE messages (
    id UUID PRIMARY KEY,
    conversation_id UUID NOT NULL REFERENCES conversations(id),
    sender_id UUID NOT NULL REFERENCES users(id),
    recipient_id UUID NOT NULL REFERENCES users(id),
    content TEXT NOT NULL,
    attachment_url VARCHAR(500),
    read BOOLEAN DEFAULT false,
    read_at TIMESTAMP,
    deleted BOOLEAN DEFAULT false,
    deleted_at TIMESTAMP,
    sent_at TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_conversation FOREIGN KEY (conversation_id) REFERENCES conversations(id),
    CONSTRAINT fk_sender FOREIGN KEY (sender_id) REFERENCES users(id),
    CONSTRAINT fk_recipient FOREIGN KEY (recipient_id) REFERENCES users(id)
);

CREATE INDEX idx_messages_conversation ON messages(conversation_id, sent_at);
CREATE INDEX idx_messages_recipient_read ON messages(recipient_id, read) WHERE read = false;
CREATE INDEX idx_messages_sent_at ON messages(sent_at DESC);

-- Unread count function
CREATE OR REPLACE FUNCTION count_unread_messages(user_id UUID, conv_id UUID)
RETURNS INTEGER AS $$
BEGIN
    RETURN (
        SELECT COUNT(*)
        FROM messages
        WHERE conversation_id = conv_id
            AND recipient_id = user_id
            AND read = false
            AND deleted = false
    );
END;
$$ LANGUAGE plpgsql;
```

**Conversation Query (Optimized):**

```sql
SELECT
    c.id,
    c.participant1_id,
    c.participant2_id,
    c.last_message_at,
    -- Get other participant info
    CASE
        WHEN c.participant1_id = :userId THEN u2.id
        ELSE u1.id
    END AS other_user_id,
    CASE
        WHEN c.participant1_id = :userId THEN u2.first_name
        ELSE u1.first_name
    END AS other_user_first_name,
    -- Last message
    m.content AS last_message_content,
    m.sender_id = :userId AS last_message_sent_by_me,
    m.read AS last_message_read,
    m.sent_at AS last_message_sent_at,
    -- Unread count
    count_unread_messages(:userId, c.id) AS unread_count
FROM conversations c
JOIN users u1 ON c.participant1_id = u1.id
JOIN users u2 ON c.participant2_id = u2.id
LEFT JOIN LATERAL (
    SELECT content, sender_id, read, sent_at
    FROM messages
    WHERE conversation_id = c.id
        AND deleted = false
    ORDER BY sent_at DESC
    LIMIT 1
) m ON true
WHERE c.participant1_id = :userId
   OR c.participant2_id = :userId
ORDER BY c.last_message_at DESC;
```

```sql
SELECT
    m.*,
    c.participant1_id,
    c.participant2_id,
    u.first_name AS sender_first_name,

FROM messages m
```

## 3. WebSocket Client Integration

### 3.1 JavaScript Client Example

```javascript
// Connect to WebSocket
const socket = new SockJS("https://api.meslektas.com/ws");
const stompClient = Stomp.over(socket);

stompClient.connect(
  { Authorization: `Bearer ${accessToken}` },
  (frame) => {
    console.log("Connected:", frame);

    // Subscribe to private messages
    stompClient.subscribe("/user/queue/messages", (message) => {
      const messageData = JSON.parse(message.body);
      displayMessage(messageData);
    });

    // Subscribe to typing indicators
    stompClient.subscribe("/user/queue/typing", (notification) => {
      const typingData = JSON.parse(notification.body);
      showTypingIndicator(typingData);
    });

    // Subscribe to read receipts
    stompClient.subscribe("/user/queue/read", (receipt) => {
      const readData = JSON.parse(receipt.body);
      updateReadStatus(readData);
    });
  },
  (error) => {
    console.error("WebSocket error:", error);
  }
);

// Send message
function sendMessage(conversationId, content) {
  stompClient.send(
    "/app/chat.send",
    {},
    JSON.stringify({
      conversationId: conversationId,
      content: content,
    })
  );
}

// Send typing indicator
let typingTimeout;
function notifyTyping(conversationId) {
  clearTimeout(typingTimeout);

  stompClient.send(
    "/app/chat.typing",
    {},
    JSON.stringify({
      conversationId: conversationId,
      typing: true,
    })
  );

  // Auto-stop after 3 seconds
  typingTimeout = setTimeout(() => {
    stompClient.send(
      "/app/chat.typing",
      {},
      JSON.stringify({
        conversationId: conversationId,
        typing: false,
      })
    );
  }, 3000);
}
```
