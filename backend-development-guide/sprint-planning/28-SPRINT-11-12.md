@Service
public class AutomatedModerationService {

    private static final List<String> BLACKLIST_KEYWORDS = List.of(
        "spam", "scam", "fake", "buy followers"
        // ... more keywords
    );

    private static final Pattern URL_PATTERN =
        Pattern.compile("https?://[^\\s]+");

    public ModerationScore scoreContent(String content) {
        int score = 0;

        // Check blacklist keywords
        for (String keyword : BLACKLIST_KEYWORDS) {
            if (content.toLowerCase().contains(keyword)) {
                score += 30;
            }
        }

        // Check excessive URLs (spam indicator)
        Matcher matcher = URL_PATTERN.matcher(content);
        int urlCount = 0;
        while (matcher.find()) urlCount++;
        if (urlCount > 3) {
            score += 20;
        }

        // Check excessive caps (shouting)
        long capsCount = content.chars()
            .filter(Character::isUpperCase)
            .count();
        if (capsCount > content.length() * 0.5) {
            score += 15;
        }

        // Check repeated characters (spam indicator)
        if (content.matches(".*([a-zA-Z])\\1{4,}.*")) {
            score += 10;
        }

        return new ModerationScore(
            score,
            score >= 50 ? RiskLevel.HIGH :
            score >= 30 ? RiskLevel.MEDIUM :
            RiskLevel.LOW
        );
    }

    public boolean shouldAutoFlag(ModerationScore score) {
        return score.getRiskLevel() == RiskLevel.HIGH;
    }
}
```

@Service
public class ReportContentCommandHandler {

    public ContentReportId handle(ReportContentCommand command) {
        // Validate content exists
        validateContentExists(command.type(), command.contentId());

        // Check for duplicate report
        if (reportRepository.existsByReporterAndContent(
            command.reporterId(),
            command.contentId()
        )) {
            throw new DuplicateReportException();
        }

        // Create report
        ContentReport report = ContentReport.create(
            command.reporterId(),
            command.type(),
            command.contentId(),
            command.reason(),
            command.description()
        );

        // Check if should auto-escalate
        int reportCount = reportRepository
            .countByContentId(command.contentId());

        if (reportCount >= 5) {
            report.escalate();
        }

        reportRepository.save(report);
        return report.getId();
    }
}
```

@Service
public class ModerationActionService {

    public void executeDecision(
        ContentReport report,
        ModerationDecision decision,
        UserId moderatorId
    ) {
        switch (decision) {
            case APPROVE_CONTENT -> {
                report.resolve(moderatorId, "No violation found");
            }

            case REMOVE_CONTENT -> {
                removeContent(report.getContentType(), report.getContentId());
                report.resolve(moderatorId, "Content removed");
                notifyContentAuthor(report, decision);
            }

            case WARN_USER -> {
                User user = getContentAuthor(report);
                user.receiveWarning(report.getReason());
                notifyUser(user, "Warning received");
            }

            case SUSPEND_USER -> {
                User user = getContentAuthor(report);
                user.suspend(Duration.ofDays(7), report.getReason());
                notifyUser(user, "Account suspended for 7 days");
            }

            case BAN_USER -> {
                User user = getContentAuthor(report);
                user.ban(report.getReason());
                notifyUser(user, "Account permanently banned");
            }
        }
    }
}

public class ContentModerationEventHandler {

    @EventListener
    @Async
    public void handlePostCreated(PostCreatedEvent event) {
        Post post = postRepository.findById(event.postId());

        // Run automated moderation
        ModerationScore score = automatedModerationService
            .scoreContent(post.getContent().getValue());

        if (score.getRiskLevel() == RiskLevel.HIGH) {
            // Auto-flag for review
            ContentReport report = ContentReport.createAutomatic(
                event.postId(),
                ReportType.POST,
                ReportReason.SPAM,
                "Automated filter: high risk score " + score.getScore()
            );

            reportRepository.save(report);

            // Optionally: auto-hide content pending review
            post.hideUntilReview();
        }

        // Log for analytics
        moderationMetrics.recordScore(score);
    }
}

          docker tag meslektas-backend:latest $ECR_REGISTRY/meslektas-backend:latest

public class SentryConfig {

        SentryOptions options = new SentryOptions();
        options.setDsn(System.getenv("SENTRY_DSN"));
        options.setEnvironment("production");
        options.setRelease("meslektas-backend@" + getVersion());
        options.setTracesSampleRate(0.2); // 20% of transactions

        // Filter errors
        options.setBeforeSend((event, hint) -> {

                return null;
            }
            return event;
        });
