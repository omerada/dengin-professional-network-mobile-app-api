```java
public class EmailNotificationService {

    private final SesClient sesClient;
    private final EmailTemplateRenderer templateRenderer;

    public void sendEmail(
        String recipient,
        String subject,
        String templateName,
        Map<String, Object> templateData
    ) {
        String htmlBody = templateRenderer.render(templateName, templateData);

        SendEmailRequest request = SendEmailRequest.builder()
            .source("noreply@meslektas.com")
            .destination(Destination.builder()
                .toAddresses(recipient)
                .build())
            .message(Message.builder()
                .subject(Content.builder().data(subject).build())
                .body(Body.builder()
                    .html(Content.builder().data(htmlBody).build())
                    .build())
                .build())
            .build();

        try {
            sesClient.sendEmail(request);
        } catch (SesException e) {
            // Retry logic
            handleEmailFailure(recipient, subject, templateData, e);
        }
    }
}
```

```java
public class NotificationPreferences {

    private Map<NotificationType, Set<DeliveryChannel>> preferences;

    public static NotificationPreferences defaults() {
        return new NotificationPreferences(
            Map.of(
                NotificationType.NEW_MESSAGE,
                    Set.of(DeliveryChannel.IN_APP, DeliveryChannel.PUSH),
                NotificationType.POST_LIKED,
                    Set.of(DeliveryChannel.IN_APP),
                NotificationType.POST_COMMENTED,
                    Set.of(DeliveryChannel.IN_APP, DeliveryChannel.EMAIL),
                NotificationType.NEW_FOLLOWER,
                    Set.of(DeliveryChannel.IN_APP, DeliveryChannel.EMAIL),
                NotificationType.VERIFICATION_APPROVED,
                    Set.of(DeliveryChannel.IN_APP, DeliveryChannel.EMAIL)
            )
        );
    }

    public boolean isEnabled(NotificationType type, DeliveryChannel channel) {
        return preferences.getOrDefault(type, Set.of()).contains(channel);
    }

    public void enable(NotificationType type, DeliveryChannel channel) {
        preferences.computeIfAbsent(type, k -> new HashSet<>()).add(channel);
    }

    public void disable(NotificationType type, DeliveryChannel channel) {
        preferences.getOrDefault(type, Set.of()).remove(channel);
    }
}
```

```java
                    .setBody(body)
                    .build()
            )
            .build();
```

```sql
    AND expires_at > CURRENT_TIMESTAMP
GROUP BY recipient_id;

CREATE UNIQUE INDEX idx_unread_counts_recipient
ON notification_unread_counts(recipient_id);

-- Refresh every 5 minutes
REFRESH MATERIALIZED VIEW CONCURRENTLY notification_unread_counts;
```

**Cleanup Job:**

```java
        Instant now = Instant.now();
        int deleted = notificationRepository.deleteExpired(now);
        log.info("Deleted {} expired notifications", deleted);
    }
}
```

**Analytics Query:**

```sql
SELECT
    type,
    COUNT(*) as total,
    COUNT(CASE WHEN read = true THEN 1 END) as read_count,
    ROUND(COUNT(CASE WHEN read = true THEN 1 END)::numeric / COUNT(*) * 100, 2) as read_rate,
    AVG(EXTRACT(EPOCH FROM (read_at - created_at))) / 60 as avg_time_to_read_minutes
```
